	join ships s on c.class = s.CLASS
	left outer join outcomes o on s.NAME = o.SHIP
where o.SHIP is null

-- or alternativly
select c.country, s.name
from classes  c
	join ships s on c.class = s.class
where s.name not in (select ship from outcomes)


select c.country ,s.name
from outcomes o 
	right outer join ships s on o.ship = s.name
	join classes c on c.class = s.class

select l.ID, r.ID
from Seat bl
	right outer join seat l on bl.id = l.id -1
	join seat r on l.id = r.id -- <=
	left outer join seat ar on r.id +1 = ar.id 
where 1 = all (select isFree from Seat where Id between l.ID and r.ID)
	and (bl.isFree = 0 or ar.isFree is null) and (ar.isFree = 0 or ar.isFree is null)
