

-- movies with length bigger than starwars


use movies

SELECT m.*
from MOVIE m
WHERE m.LENGTH > (SELECT m1.LENGTH FROM MOVIE m1 WHERE m1.TITLE = 'Star Wars' AND m1.YEAR = 1977)

-- movie names, actress, birthday
use movies

select *
from moviestar ms
	join starsin si on si.STARNAME = ms.NAME
where ms.gender = 'F'

from starsin si

		from moviestar ms

-- all pc's, laptop, printer from B

use pc

select l.model, l.price, 'laptop' as type 
from laptop l
	join product p on l.model = p.model
union
select l.model, l.price, 'pc' as type 
from pc l
	join product p on l.model = p.model
union 
select l.model, l.price, 'printer' as type 
from printer l
	join product p on l.model = p.model



select *
	from (
select l.model, l.price, 'laptop' as type 
from laptop l
union
select l.model, l.price, 'pc' as type 
from pc l
union 
select l.model, l.price, 'printer' as type 
from printer l
	) u
	join product p on p.model = u.model 
where p.maker = 'B'

				from starsin si
