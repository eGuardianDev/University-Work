select s.name,
	   s.address,
	   count(distinct incolor),
	   string_agg(title + ':' + cast (Year as varchar(max)), ', '),
	   coalesce(sum(m.length),0)
from studio s
	left outer join movie m on s.name = m.studioname 
group by s.name, s.address


select s.name,
	   s.address,
	   count(*),
	   count(length),
	   count(distinct incolor),
	   count(distinct PRODUCERC#),
	   string_agg(title + ':' + cast (Year as varchar(max)), ', '),
	   coalesce(sum(m.length),0)
from studio s
	left outer join movie m on s.name = m.studioname

-- сумарната дължина на всички филми заснети от Харисън Форд

select starname, sum(Length)
from MOVIE m 
	join starsin si on m.title = si.MOVIETITLE and m.YEAR = si.MOVIEYEAR
where si.STARNAME = 'Harrison Ford'
group by starname

select starname, sum(Length)
from MOVIE m 
	join starsin si on m.title = si.MOVIETITLE and m.YEAR = si.MOVIEYEAR
group by starname
having starname = 'Harrison Ford'

-- среден брой филми, в който са се снимали всички актьори
-- средно аритметично от броя филми за вески актьор


select avg(mc.moviecount) from
(select ms.name, count(starname) as moviecount
from moviestar ms
	left outer join starsin si on ms.name = si.starname
group by ms.name) mc


-- броя на корабите потънали в битка според класа
use ships

select *
from outcomes
where result ='sunk'


select s.class, count(o.result)
from outcomes o
	join ships s on o.ship = s.name
where result = 'sunk'
group by s.class


select c.CLASS, count(o.ship)
from classes c 
	left outer join ships s on c.class = s.class
	left outer join outcomes o on o.ship = s.name and o.result = 'sunk'
group by c.class


select c.CLASS, count(case o.result when 'sunk' then 1 else 0 end)
from classes c 
	left outer join ships s on c.class = s.class
	left outer join outcomes o on o.ship = s.name
group by c.class
